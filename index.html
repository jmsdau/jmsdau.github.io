12.Another 01Game-放弃使用qrcode，改用image模块编写程序生成二维码，程序如下:
    from PIL import Image
    MAX = 37
    pic = Image.new('RGB',(MAX*37,MAX*37))
    f = open('code.txt','r')
    str = f.read()
    i = 0
    for y in range(0,MAX*37,37):
        for x in range(0,MAX*37,37):
            if(str[i] == '0'):
                for n in range(37):
                    for j in range(37):
                        pic.putpixel([x+j,y+n],(0,0,0))
            else:
                for k in range(37):
                    for l in range(37):
                        pic.putpixel([x+l,y+k],(255,255,255))
            i = i+1

    pic.show()
    pic.save("flag.png")
    f.close()
